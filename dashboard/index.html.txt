async function atualizar() {

    try {

        let status = await fetch('/status').then(r => r.json())

        document.getElementById('rodando').innerHTML =
            status.rodando
            ? '<span class="status-run">⏳ Rodando...</span>'
            : '<span class="status-ok">✅ Parado</span>'

        document.getElementById('total').innerText = status.total || 0
        document.getElementById('atualizados').innerText = status.atualizados || 0
        document.getElementById('criados').innerText = status.criados || 0
        document.getElementById('zerados').innerText = status.zerados || 0
        document.getElementById('erros').innerText = status.erros || 0

        let logs = await fetch('/logs').then(r => r.json())

        document.getElementById('logs').innerText = logs.join("\n")

    } catch(e) {
        console.log("erro ao atualizar", e)
    }
}

function executar() {
    fetch('/executar')
}

setInterval(atualizar, 2000)

</script>
